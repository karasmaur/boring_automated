Select
ROWIDTOCHAR (MENSAGEM_OCORRENCIA_BACEN.ROWID) as CHAVE
            , MENSAGEM_BACEN.OID_MENSAGEM_BACEN
            , DECODE( MENSAGEM_BACEN.TPO_EVENTO, 'I', 'Inclusão', 'A', 'Alteração', 'E', 'Exclusão' ) as INC_ALT_EXC
            , TO_CHAR( MENSAGEM_OCORRENCIA_BACEN.DAT_OCORRENCIA, 'DD/MM/YYYY HH24:MI:SS' ) as DATA_OCORRENCIA
            , SITUACAO_BACEN.COD_SITUACAO
            , SITUACAO_BACEN.NOM_SITUACAO
            , GRUPO_SITUACAO_BACEN.COD_GRUPO_SITUACAO
            , GRUPO_SITUACAO_BACEN.NOM_GRUPO_SITUACAO
            , mensagem_ocorrencia_bacen.oid_msg_ocorrencia_bacen
            , mensagem_bacen.oid_mensagem_bacen
            , titulo_bacen.oid_operacao
          FROM TITULO_BACEN,
               MENSAGEM_BACEN,
               MENSAGEM_OCORRENCIA_BACEN,
               SITUACAO_BACEN,
               GRUPO_SITUACAO_BACEN
         WHERE TITULO_BACEN.OID_TITULO_BACEN =
               MENSAGEM_BACEN.OID_TITULO_BACEN
           AND MENSAGEM_BACEN.OID_MENSAGEM_BACEN =
               MENSAGEM_OCORRENCIA_BACEN.OID_MENSAGEM_BACEN
           AND MENSAGEM_OCORRENCIA_BACEN.OID_SITUACAO_BACEN =
               SITUACAO_BACEN.OID_SITUACAO_BACEN
           AND SITUACAO_BACEN.OID_GRUPO_SITUACAO_BACEN =
               GRUPO_SITUACAO_BACEN.OID_GRUPO_SITUACAO_BACEN
           AND MENSAGEM_OCORRENCIA_BACEN.OID_MSG_OCORRENCIA_BACEN IN
               (SELECT MAX(MENSAGEM_OCORRENCIA_BACEN.OID_MSG_OCORRENCIA_BACEN) OID_MSG_OCORRENCIA_BACEN
                  FROM TITULO_BACEN TB,
                       MENSAGEM_BACEN,
                       MENSAGEM_OCORRENCIA_BACEN
                 WHERE TB.OID_TITULO_BACEN = MENSAGEM_BACEN.OID_TITULO_BACEN
                   AND MENSAGEM_BACEN.OID_MENSAGEM_BACEN =
                       MENSAGEM_OCORRENCIA_BACEN.OID_MENSAGEM_BACEN
                   AND TB.OID_OPERACAO = TITULO_BACEN.OID_OPERACAO)
